```ts
// apps/web/src/utils.ts
import type { Action } from './types'

export function getActionsThisMonth(actions: Action[], yearMonth: string): Action[] {
  return actions.filter(a => a.date.startsWith(yearMonth))
}

export function daysSinceLastAction(actions: Action[]): number {
  if (actions.length === 0) return 0
  const sorted = [...actions].sort((a, b) => b.date.localeCompare(a.date))
  const last = new Date(sorted[0].date)
  const today = new Date()
  today.setHours(0, 0, 0, 0)
  return Math.floor((today.getTime() - last.getTime()) / (1000 * 60 * 60 * 24))
}

export function extractLastPh(actions: Action[]): string {
  const sorted = [...actions].sort((a, b) => b.date.localeCompare(a.date))
  for (const action of sorted) {
    const match = action.meta.match(/pH\s*([\d.]+)/)
    if (match) return match[1]
  }
  return '—'
}
```

```tsx
import { useState } from 'react'
import type { Action } from '../types'

const PRODUCTS = ['Brome', 'Chlore', 'pH-', 'pH+', 'Anti-algue', 'Floculant', 'Sel']
const ACTIONS = ['Nettoyage cartouche', 'Ajout de chlore', 'Ajout de brome', 'Réglage pH', 'Contre-lavage', 'Mesure pH', 'Nettoyage filtre']
const UNITS = ['g', 'ml', 'pastille', 'L']

type Props = {
  onAdd: (action: Omit<Action, 'id'>) => void
  onClose?: () => void
  mobileSheet?: boolean
}

export default function ActionForm({ onAdd, onClose, mobileSheet = false }: Props) {
  const today = new Date().toISOString().slice(0, 10)
  const [date, setDate] = useState(today)
  const [title, setTitle] = useState(ACTIONS[0])
  const [product, setProduct] = useState(PRODUCTS[0])
  const [qty, setQty] = useState('')
  const [unit, setUnit] = useState(UNITS[0])
  const [freeProduct, setFreeProduct] = useState('')
  const [notes, setNotes] = useState('')

  const handleSubmit = (e: React.FormEvent) => {
    e.preventDefault()
    const meta = [
      product && qty ? `${product} ${qty}${unit}` : '',
      freeProduct,
      notes,
    ].filter(Boolean).join(' — ')
    onAdd({ date, title, meta: meta || title })
    setQty('')
    setFreeProduct('')
    setNotes('')
    onClose?.()
  }

  const formStyle: React.CSSProperties = mobileSheet ? {
    position: 'fixed', inset: 0, borderRadius: 0, padding: '64px 22px 22px', zIndex: 30,
    background: '#0f191c', color: '#f3f6f8',

    background: '#0f191c', color: '#f3f6f8',
    borderRadius: 28, padding: 26,
    boxShadow: 'var(--shadow)',

  return (
```

### Task 10 : App.tsx — assemblage final

**Files:**

**Step 1 : Remplacer App.tsx**

```tsx
// apps/web/src/App.tsx
import { useState } from 'react'
import type { Action } from './types'
import { mockActions } from './data/mockActions'
import Topbar from './components/Topbar'
import HeroCard from './components/HeroCard'
import ActionForm from './components/ActionForm'
import InfoGrid from './components/InfoGrid'

export default function App() {
  const [actions, setActions] = useState<Action[]>(mockActions)
  const [showForm, setShowForm] = useState(false)

  const handleAdd = (newAction: Omit<Action, 'id'>) => {
    const action: Action = { id: Date.now(), ...newAction }
    setActions(prev => [action, ...prev].sort((a, b) => b.date.localeCompare(a.date)))
  }

  return (
    <div className="page">
      <Topbar />

      <section style={{
        display: 'grid',
        gridTemplateColumns: 'minmax(0, 1.1fr) minmax(0, 0.9fr)',
        gap: 24,
        alignItems: 'stretch',
      }}
        className="hero">
        <HeroCard actions={actions} onOpenForm={() => setShowForm(true)} />

        {/* Desktop: always visible */}
        <div className="form-desktop">
          <ActionForm onAdd={handleAdd} />
        </div>
      </section>

      <InfoGrid />

      {/* Mobile: bottom sheet */}
      {showForm && (
        <ActionForm
          onAdd={handleAdd}
          onClose={() => setShowForm(false)}
          mobileSheet
        />
      )}

      {/* Mobile FAB */}
      {!showForm && (
        <button
          className="btn primary open-form"
          onClick={() => setShowForm(true)}
          style={{
            position: 'fixed', bottom: 24, right: 24, zIndex: 20,
            borderRadius: 999, padding: '14px 18px',
          }}
        >
          Ajouter une action
        </button>
      )}
    </div>
  )
}
```
