```python
from sqlmodel import Field, SQLModel


class Product(SQLModel, table=True):

    action_type: str
    product_id: Optional[int] = Field(default=None, foreign_key="product.id")
    qty: str = ""
    unit: str = ""
    notes: str = ""
    created_at: datetime = Field(default_factory=datetime.utcnow)
```

```python
    {"name": "Sel", "unit_default": "kg"},

    existing = session.exec(select(Product)).all()
    existing_names = {p.name for p in existing}
```

```python
# apps/api/main.py
from contextlib import asynccontextmanager
from datetime import date, datetime
from typing import List, Optional

from fastapi import Depends, FastAPI, HTTPException
from fastapi.middleware.cors import CORSMiddleware
from pydantic import BaseModel
from sqlmodel import Session, SQLModel, select

from database import engine, get_session
from models import Action, Product
from seeds import insert_seeds


@asynccontextmanager
async def lifespan(app: FastAPI):
    SQLModel.metadata.create_all(engine)
    with Session(engine) as session:
        insert_seeds(session)
    yield


app = FastAPI(title="Pooly API", lifespan=lifespan)

app.add_middleware(
    CORSMiddleware,
    allow_origins=["*"],
    allow_credentials=True,
    allow_methods=["*"],
    allow_headers=["*"],
)


# ── Schémas compat (Phase 1) ──────────────────────────────────────────────────

class ActionInCompat(BaseModel):
    """Payload ancien format envoyé par le frontend Phase 1."""
    date: date
    title: str
    meta: str = ""


class ActionCompat(BaseModel):
    """Réponse ancien format attendu par le frontend Phase 1."""
    id: int
    date: date
    title: str
    meta: str


# ── Helpers ───────────────────────────────────────────────────────────────────

def _to_compat(action: Action, product: Optional[Product]) -> ActionCompat:
    if product:
        meta = f"{product.name} {action.qty}{action.unit}".strip()
    else:
        meta = action.notes
    return ActionCompat(id=action.id, date=action.date, title=action.action_type, meta=meta)


# ── Endpoints ─────────────────────────────────────────────────────────────────

@app.get("/health")
def health():
    return {"status": "ok"}


@app.get("/products")
def list_products(session: Session = Depends(get_session)):
    return session.exec(select(Product)).all()


@app.get("/actions", response_model=List[ActionCompat])
def list_actions(session: Session = Depends(get_session)):
    actions = session.exec(select(Action).order_by(Action.date.desc())).all()
    result = []
    for a in actions:
        product = session.get(Product, a.product_id) if a.product_id else None
        result.append(_to_compat(a, product))
    return result


@app.post("/actions", response_model=ActionCompat)
def create_action(payload: ActionInCompat, session: Session = Depends(get_session)):
    action = Action(
        date=payload.date,
        action_type=payload.title,
        notes=payload.meta,
        created_at=datetime.utcnow(),
    )
    session.add(action)
    session.commit()
    session.refresh(action)
    return _to_compat(action, None)


@app.delete("/actions/{action_id}", status_code=204)
def delete_action(action_id: int, session: Session = Depends(get_session)):
    action = session.get(Action, action_id)
    if not action:
        raise HTTPException(status_code=404, detail="Action introuvable")
    session.delete(action)
    session.commit()
```

```typescript
// apps/web/src/utils.ts
import type { Action } from './types'

export function getActionsThisMonth(actions: Action[], yearMonth: string): Action[] {
  return actions.filter(a => a.date.startsWith(yearMonth))
}

export function daysSinceLastAction(actions: Action[]): number {
  if (actions.length === 0) return 0
  const sorted = [...actions].sort((a, b) => b.date.localeCompare(a.date))
  const [year, month, day] = sorted[0].date.split('-').map(Number)
  const lastUtc = Date.UTC(year, month - 1, day)
  const now = new Date()
  const todayUtc = Date.UTC(now.getUTCFullYear(), now.getUTCMonth(), now.getUTCDate())
  return Math.floor((todayUtc - lastUtc) / (1000 * 60 * 60 * 24))
}

export function extractLastPh(actions: Action[]): string {
  const sorted = [...actions].sort((a, b) => b.date.localeCompare(a.date))
  for (const action of sorted) {
    const match = action.notes.match(/pH\s*([\d.]+)/)
    if (match) return match[1]
  }
  return '—'
}
```

```typescript
// apps/web/src/components/ActionEntry.tsx
import type { Action, Product } from '../types'

type Props = {
  action: Action
  products: Product[]
}

function formatDate(dateStr: string): string {
  const [, month, day] = dateStr.split('-')
  return `${day}/${month}`
}

export default function ActionEntry({ action, products }: Props) {
  const product = products.find(p => p.id === action.product_id) ?? null
  const subtitle = product
    ? `${product.name}${action.qty ? ` ${action.qty}${action.unit}` : ''}`.trim()
    : action.notes

  return (
    <div style={{ display: 'grid', gridTemplateColumns: '80px minmax(0,1fr)', gap: 12, alignItems: 'center' }}>
      <div style={{ fontWeight: 600, color: 'var(--sea-2)', fontSize: 13 }}>{formatDate(action.date)}</div>
      <div style={{ background: '#fff', borderRadius: 18, padding: '12px 14px', border: '1px solid var(--line)', display: 'grid', gap: 6 }}>
        <div style={{ fontWeight: 600, fontSize: 14 }}>{action.action_type}</div>
        <div style={{ color: 'var(--muted)', fontSize: 12 }}>{subtitle}</div>
      </div>
    </div>
  )
}
```

```typescript
// apps/web/src/App.tsx
import { useState, useEffect } from 'react'
import type { Action, Product } from './types'
import Topbar from './components/Topbar'
import HeroCard from './components/HeroCard'
import ActionForm from './components/ActionForm'
import InfoGrid from './components/InfoGrid'

export default function App() {
  const [actions, setActions] = useState<Action[]>([])
  const [products, setProducts] = useState<Product[]>([])
  const [loading, setLoading] = useState(true)
  const [error, setError] = useState<string | null>(null)
  const [showForm, setShowForm] = useState(false)

  useEffect(() => {
    Promise.all([
      fetch('/api/actions').then(r => r.json()),
      fetch('/api/products').then(r => r.json()),
    ])
      .then(([actionsData, productsData]) => {
        setActions(actionsData)
        setProducts(productsData)
      })
      .catch(() => setError('Impossible de charger les données'))
      .finally(() => setLoading(false))
  }, [])

  const handleAdd = async (newAction: Omit<Action, 'id' | 'created_at'>) => {
    const res = await fetch('/api/actions', {
      method: 'POST',
      headers: { 'Content-Type': 'application/json' },
      body: JSON.stringify(newAction),
    })
    const action: Action = await res.json()
    setActions(prev => [action, ...prev].sort((a, b) => b.date.localeCompare(a.date)))
  }

  if (loading) return <div className="page"><p style={{ padding: 24 }}>Chargement…</p></div>
  if (error) return <div className="page"><p style={{ padding: 24, color: 'var(--danger)' }}>{error}</p></div>

  return (
    <div className="page">
      <Topbar />

      <section
        style={{
          display: 'grid',
          gridTemplateColumns: 'minmax(0, 1.1fr) minmax(0, 0.9fr)',
          gap: 24,
          alignItems: 'stretch',
        }}
        className="hero"
      >
        <HeroCard actions={actions} products={products} onOpenForm={() => setShowForm(true)} />

        <div className="form-desktop">
          <ActionForm onAdd={handleAdd} products={products} />
        </div>
      </section>

      <InfoGrid />

      {showForm && (
        <ActionForm onAdd={handleAdd} products={products} onClose={() => setShowForm(false)} mobileSheet />
      )}

      {!showForm && (
        <button
          className="btn primary open-form"
          onClick={() => setShowForm(true)}
          style={{
            position: 'fixed',
            bottom: 24,
            right: 24,
            zIndex: 20,
            borderRadius: 999,
            padding: '14px 18px',
          }}
        >
          Ajouter une action
        </button>
      )}
    </div>
  )
}
```

```python
# apps/api/main.py
from contextlib import asynccontextmanager
from datetime import date, datetime
from typing import List, Optional

from fastapi import Depends, FastAPI, HTTPException
from fastapi.middleware.cors import CORSMiddleware
from pydantic import BaseModel
from sqlmodel import Session, SQLModel, select

from database import engine, get_session
from models import Action, Product
from seeds import insert_seeds


@asynccontextmanager
async def lifespan(app: FastAPI):
    SQLModel.metadata.create_all(engine)
    with Session(engine) as session:
        insert_seeds(session)
    yield


app = FastAPI(title="Pooly API", lifespan=lifespan)

app.add_middleware(
    CORSMiddleware,
    allow_origins=["*"],
    allow_credentials=True,
    allow_methods=["*"],
    allow_headers=["*"],
)


# ── Schémas ───────────────────────────────────────────────────────────────────

class ActionIn(BaseModel):
    date: date
    action_type: str
    product_id: Optional[int] = None
    qty: str = ""
    unit: str = ""
    notes: str = ""


class ActionOut(BaseModel):
    id: int
    date: date
    action_type: str
    product_id: Optional[int]
    qty: str
    unit: str
    notes: str
    created_at: datetime


# ── Endpoints ─────────────────────────────────────────────────────────────────

@app.get("/health")
def health():
    return {"status": "ok"}


@app.get("/products")
def list_products(session: Session = Depends(get_session)):
    return session.exec(select(Product)).all()


@app.get("/actions", response_model=List[ActionOut])
def list_actions(session: Session = Depends(get_session)):
    return session.exec(select(Action).order_by(Action.date.desc())).all()


@app.post("/actions", response_model=ActionOut)
def create_action(payload: ActionIn, session: Session = Depends(get_session)):
    action = Action(
        date=payload.date,
        action_type=payload.action_type,
        product_id=payload.product_id,
        qty=payload.qty,
        unit=payload.unit,
        notes=payload.notes,
        created_at=datetime.utcnow(),
    )
    session.add(action)
    session.commit()
    session.refresh(action)
    return action


@app.delete("/actions/{action_id}", status_code=204)
def delete_action(action_id: int, session: Session = Depends(get_session)):
    action = session.get(Action, action_id)
    if not action:
        raise HTTPException(status_code=404, detail="Action introuvable")
    session.delete(action)
    session.commit()
```
