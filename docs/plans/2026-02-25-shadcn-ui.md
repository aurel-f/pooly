# Refonte UI shadcn/ui Implementation Plan

> **For Claude:** REQUIRED SUB-SKILL: Use superpowers:executing-plans to implement this plan task-by-task.

**Goal:** Migrer les composants interactifs vers shadcn/ui (Tailwind 4 + Radix UI) avec un nouveau layout une colonne (stats + historique), formulaire en Dialog.

**Architecture:** Installation de Tailwind 4 via `@tailwindcss/vite`, setup shadcn/ui via CLI, migration composant par composant sans toucher à la logique métier ni aux appels API. Le CSS custom est conservé pour le layout global (`.page`, media queries) et les variables de couleurs.

**Tech Stack:** Tailwind CSS 4, shadcn/ui (default theme, neutral base), Radix UI, lucide-react, class-variance-authority, clsx, tailwind-merge

**Step 1 : Réécrire `apps/web/src/components/ActionForm.tsx`**

```tsx
import { useState } from 'react'
import type { Action, Product } from '../types'
import { Button } from '@/components/ui/button'
import { Input } from '@/components/ui/input'
import { Label } from '@/components/ui/label'
import { Textarea } from '@/components/ui/textarea'
import {
  Select,
  SelectContent,
  SelectItem,
  SelectTrigger,
  SelectValue,
} from '@/components/ui/select'

const ACTION_TYPES = [
  'Nettoyage cartouche',
  'Ajout de chlore',
  'Ajout de brome',
  'Réglage pH',
  'Contre-lavage',
  'Mesure pH',
  'Nettoyage filtre',
]
const UNITS = ['g', 'ml', 'pastille', 'L']

type Props = {
  onAdd: (action: Omit<Action, 'id' | 'created_at' | 'user_id'>) => void
  products: Product[]
  onClose?: () => void
}

export default function ActionForm({ onAdd, products, onClose }: Props) {
  const today = new Date().toISOString().slice(0, 10)
  const [date, setDate] = useState(today)
  const [actionType, setActionType] = useState(ACTION_TYPES[0])
  const [productId, setProductId] = useState<number | null>(null)
  const [qty, setQty] = useState('')
  const [unit, setUnit] = useState(UNITS[0])
  const [notes, setNotes] = useState('')

  const handleSubmit = (e: React.FormEvent) => {
    e.preventDefault()
    onAdd({ date, action_type: actionType, product_id: productId, qty, unit, notes })
    setQty('')
    setNotes('')
    onClose?.()
  }

  return (
    <form onSubmit={handleSubmit} className="grid gap-4">
      <div className="grid gap-1.5">
        <Label htmlFor="date">Date</Label>
        <Input
          id="date"
          type="date"
          value={date}
          onChange={e => setDate(e.target.value)}
          required
        />
      </div>

      <div className="grid gap-1.5">
        <Label htmlFor="action-type">Action</Label>
        <Select value={actionType} onValueChange={setActionType}>
          <SelectTrigger id="action-type">
            <SelectValue />
          </SelectTrigger>
          <SelectContent>
            {ACTION_TYPES.map(a => (
              <SelectItem key={a} value={a}>{a}</SelectItem>
            ))}
          </SelectContent>
        </Select>
      </div>

      <div className="grid grid-cols-2 gap-3">
        <div className="grid gap-1.5">
          <Label htmlFor="product">Produit</Label>
          <Select
            value={productId !== null ? String(productId) : ''}
            onValueChange={v => setProductId(v ? Number(v) : null)}
          >
            <SelectTrigger id="product">
              <SelectValue placeholder="Aucun" />
            </SelectTrigger>
            <SelectContent>
              <SelectItem value="">Aucun</SelectItem>
              {products.map(p => (
                <SelectItem key={p.id} value={String(p.id)}>{p.name}</SelectItem>
              ))}
            </SelectContent>
          </Select>
        </div>
        <div className="grid gap-1.5">
          <Label htmlFor="qty">Quantité</Label>
          <Input
            id="qty"
            type="number"
            value={qty}
            onChange={e => setQty(e.target.value)}
            placeholder="60"
          />
        </div>
      </div>

      <div className="grid gap-1.5">
        <Label htmlFor="unit">Unité</Label>
        <Select value={unit} onValueChange={setUnit}>
          <SelectTrigger id="unit">
            <SelectValue />
          </SelectTrigger>
          <SelectContent>
            {UNITS.map(u => (
              <SelectItem key={u} value={u}>{u}</SelectItem>
            ))}
          </SelectContent>
        </Select>
      </div>

      <div className="grid gap-1.5">
        <Label htmlFor="notes">Notes</Label>
        <Textarea
          id="notes"
          value={notes}
          onChange={e => setNotes(e.target.value)}
          placeholder="Ex: eau claire, niveau OK"
        />
      </div>

      <Button type="submit" className="w-full">Enregistrer</Button>
    </form>
  )
}
```

```tsx
import type { Action, Product } from '../types'
import { Card, CardContent } from '@/components/ui/card'
import { Badge } from '@/components/ui/badge'

type Props = {
  action: Action
  products: Product[]
}

function formatDate(dateStr: string): string {
  const [, month, day] = dateStr.split('-')
  return `${day}/${month}`
}

function formatMeta(action: Action, products: Product[]): string {
  const product = products.find(p => p.id === action.product_id)
  if (!product) return action.notes || '—'
  const qty = action.qty ? ` ${action.qty}` : ''
  const unit = action.unit ? action.unit : ''
  return `${product.name}${qty}${unit}`.trim() || '—'
}

export default function ActionEntry({ action, products }: Props) {
  return (
    <div className="grid gap-2" style={{ gridTemplateColumns: '52px minmax(0,1fr)' }}>
      <div className="text-xs font-semibold text-muted-foreground pt-3">
        {formatDate(action.date)}
      </div>
      <Card>
        <CardContent className="pt-3 pb-3 px-4 flex items-start justify-between gap-2">
          <div>
            <Badge variant="outline" className="mb-1 text-xs">{action.action_type}</Badge>
            <div className="text-xs text-muted-foreground">{formatMeta(action, products)}</div>
          </div>
        </CardContent>
      </Card>
    </div>
  )
}
```
